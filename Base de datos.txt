CREATE DATABASE PROTOTIPO;

USE PROTOTIPO;

CREATE TABLE TUSUARIA(
IDUSUARIA INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
CORREO VARCHAR(60),
PASSWORD VARCHAR(120),
NOMBRE VARCHAR(60),
APELLIDOS VARCHAR(60),
FECHA_NACIMIENTO DATE,
PROVINCIA VARCHAR(40),
SECTOR_PROFESIONAL VARCHAR(60),
INTERESES VARCHAR(500),
ALUMNA TINYINT,
FORMADORA TINYINT);

CREATE TABLE TEVENTO(
IDEVENTO INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
TIPO_DE_FORMACION VARCHAR(30),
FORMACION VARCHAR(1000),
REUNION VARCHAR(250));



CREATE TABLE TREGISTROEVENTO(
IDREGISTRO INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
IDUSUARIA INT NOT NULL,
IDEVENTO INT NOT NULL,
NOMBRE_PROFESOR VARCHAR(60),
FECHA_REGISTRO DATE);

ALTER TABLE TREGISTROEVENTO ADD FOREIGN KEY(IDUSUARIA) REFERENCES TUSUARIA(IDUSUARIA);
ALTER TABLE TREGISTROEVENTO ADD FOREIGN KEY(IDEVENTO) REFERENCES TEVENTO(IDEVENTO);
